📋 DSA Pattern Cheat Sheet

1. Sliding Window
Use when: contiguous subarray/substring, find max/min/length with a condition
l = 0 for r in range(len(arr)): # expand window: process arr[r] while window_invalid: # shrink: process arr[l] l += 1 ans = max(ans, r - l + 1)
2. Two Pointers
Use when: sorted array, find pair with condition, palindrome check
l, r = 0, len(arr) - 1 while l < r: if arr[l] + arr[r] == target: return [l, r] elif arr[l] + arr[r] < target: l += 1 else: r -= 1
3. Fast & Slow Pointers
Use when: linked list cycle, find middle, duplicate detection
slow = fast = head while fast and fast.next: slow = slow.next fast = fast.next.next if slow == fast: return True # cycle return False
4. Binary Search
Use when: sorted array, find target, find boundary condition
l, r = 0, len(arr) - 1 while l <= r: m = (l + r) // 2 if arr[m] == target: return m elif arr[m] < target: l = m + 1 else: r = m - 1 return -1
5. BFS (Level Order)
Use when: shortest path, level-by-level traversal, graph/tree
from collections import deque q = deque([start]) visited = {start} while q: node = q.popleft() for neighbor in graph[node]: if neighbor not in visited: visited.add(neighbor) q.append(neighbor)
6. DFS (Recursive)
Use when: explore all paths, tree traversal, connected components
def dfs(node, visited): if node in visited: return visited.add(node) for neighbor in graph[node]: dfs(neighbor, visited)
7. Dynamic Programming
Use when: overlapping subproblems, optimal substructure, count ways
# Top-down (memoization) from functools import lru_cache @lru_cache(maxsize=None) def dp(i): return dp(i-1) + dp(i-2) # Bottom-up (tabulation) dp = [0] * (n+1) dp[0], dp[1] = 1, 1 for i in range(2, n+1): dp[i] = dp[i-1] + dp[i-2]
8. Backtracking
Use when: generate all combinations/permutations, constraint satisfaction
def backtrack(path, choices): if base_case: result.append(path[:]) for choice in choices: path.append(choice) backtrack(path, updated_choices) path.pop() # undo
9. Heap / Priority Queue
Use when: K largest/smallest, merge K sorted lists, scheduling
import heapq heap = [] heapq.heappush(heap, val) # push heapq.heappop(heap) # pop min # For max heap: push -val # K largest: heapq.nlargest(k, arr)
10. HashMap / HashSet
Use when: frequency count, two-sum type, dedup, O(1) lookup
freq = {} for x in arr: freq[x] = freq.get(x, 0) + 1 # Two sum pattern seen = {} for i, x in enumerate(arr): if target - x in seen: return [seen[target-x], i] seen[x] = i

🐍 Python Quick Reference

List Operations
a = [1,2,3] a.append(4) # [1,2,3,4] a.pop() # removes last → 4 a.insert(0, 0) # [0,1,2,3] a.sort() # in-place sort sorted(a) # returns new sorted list a[::-1] # reverse a[1:3] # slice [2,3] len(a) # length sum(a) # sum min(a), max(a) # min, max
List Comprehension
[x*2 for x in range(5)] # [0,2,4,6,8] [x for x in arr if x > 0] # filter positives [x**2 for x in range(1,6)] # [1,4,9,16,25] matrix = [[0]*3 for _ in range(3)] # 3x3 grid
Dictionary
d = {} d['key'] = 'val' d.get('key', 'default') # safe get 'key' in d # check exists d.keys(), d.values(), d.items() # views from collections import defaultdict d = defaultdict(int) # default 0 d = defaultdict(list) # default [] from collections import Counter Counter([1,1,2,3]) # {1:2, 2:1, 3:1}
String Operations
s = "hello world" s.split() # ['hello', 'world'] ' '.join(['a','b']) # 'a b' s.strip() # remove whitespace s.lower(), s.upper() s.replace('l','L') s[::-1] # reverse string ord('a') = 97 # char to ASCII chr(97) = 'a' # ASCII to char s.isalnum() # alphanumeric check
Useful Built-ins
enumerate(arr) # (index, value) pairs zip(a, b) # pair two lists range(n) # 0 to n-1 range(a, b, step) # a to b-1, step map(func, arr) # apply func to all filter(func, arr) # keep where func=True abs(-5) = 5 float('inf') # infinity float('-inf') # negative infinity
Deque (Use for BFS / Sliding Window)
from collections import deque q = deque() q.append(1) # push right q.appendleft(0) # push left q.pop() # pop right q.popleft() # pop left O(1) ← use this!
Sorting Tricks
arr.sort(key=lambda x: x[1]) # sort by second element arr.sort(key=lambda x: -x) # sort descending arr.sort(key=lambda x: (x[0],-x[1])) # multi-key sort
